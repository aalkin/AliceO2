44  {
45  mOutputs = specs::buildOutputs(outputSpecs);
46 
47  if (useMC) {
48  mFillLabels = [](size_t inIdx, size_t outIdx, const o2::dataformats::MCTruthContainer<MCLabel>* inMCContainer, o2::dataformats::MCTruthContainer<MCLabel>& outMCContainer, const ColumnData& col) {
49  auto labels = inMCContainer->getLabels(inIdx);
50  for (auto& label : labels) {
51  if (label.getDEId() == col.deId && label.getColumnId() == col.columnId) {
52  bool hasLabel = false;
53  for (int istrip = label.getFirstStrip(), last = label.getLastStrip(); istrip <= last; ++istrip) {
54  if ((label.getCathode() == 1 && col.isNBPStripFired(istrip)) || (label.getCathode() == 0 && col.isBPStripFired(label.getStripInLine(istrip), label.getLine(istrip)))) {
55  hasLabel = true;
56  break;
57  }
58  }
59  if (hasLabel) {
60  outMCContainer.addElement(outIdx, label);
61  }
62  }
63  }
64  };
65  }

81  void run(of::ProcessingContext& pc)
82  {
83  const auto badChannels = pc.inputs().get<std::vector<ColumnData>*>("mid_bad_channels");
84 
85  auto data = specs::getData(pc, "mid_filter_in", EventType::Standard);
86  auto inROFRecords = specs::getRofs(pc, "mid_filter_in", EventType::Standard);
87 
88  std::unique_ptr<const o2::dataformats::MCTruthContainer<MCLabel>> inMCContainer = mUseMC ? specs::getLabels(pc, "mid_filter_in") : nullptr;
89 
90  auto& maskedData = pc.outputs().make<std::vector<ColumnData>>(of::OutputRef{"mid_filter_out_0"});
91  auto& maskedRofs = pc.outputs().make<std::vector<ROFRecord>>(of::OutputRef{"mid_filter_out_rof_0"});
92 
93  maskedData.reserve(data.size());
94  maskedRofs.reserve(inROFRecords.size());
95 
96  o2::dataformats::MCTruthContainer<MCLabel> outMCContainer;
97 
98  for (auto& rof : inROFRecords) {
99  auto firstEntry = maskedData.size();
100  for (auto dataIt = data.begin() + rof.firstEntry, end = data.begin() + rof.getEndIndex(); dataIt != end; ++dataIt) {
101  auto col = *dataIt;
102  if (mMasksHandler.applyMask(col)) {
103  // Data are not fully masked
104  maskedData.emplace_back(col);
105  auto inIdx = std::distance(data.begin(), dataIt);
106  mFillLabels(inIdx, maskedData.size() - 1, inMCContainer.get(), outMCContainer, col);
107  }
108  }
109  auto nEntries = maskedData.size() - firstEntry;
110  if (nEntries > 0) {
111  maskedRofs.emplace_back(rof);
112  maskedRofs.back().firstEntry = firstEntry;
113  maskedRofs.back().nEntries = nEntries;
114  }
115  }
116 
117  if (mUseMC) {
118  pc.outputs().snapshot(mOutputs[2], outMCContainer);
119  }
120  }
121 
122  private:
123  ChannelMasksHandler mMasksHandler{};
124  bool mUseMC{false};
125  std::vector<of::Output> mOutputs;
126  std::function<void(size_t, size_t, const o2::dataformats::MCTruthContainer<MCLabel>*, o2::dataformats::MCTruthContainer<MCLabel>&, const ColumnData& col)> mFillLabels{[](size_t, size_t, const o2::dataformats::MCTruthContainer<MCLabel>*, o2::dataformats::MCTruthContainer<MCLabel>&, const ColumnData&) {}};
