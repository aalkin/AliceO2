39 void GPUDisplayMagneticField::generateSeedPoints(std::size_t count)
40 {
41  std::mt19937 rng(0xDEADBEEF);
42 
43  // TODO: fetch these values from somewhere?
44  std::uniform_int_distribution<int> generator_x(-512, 512);
45  std::uniform_int_distribution<int> generator_y(-512, 512);
46  std::uniform_int_distribution<int> generator_z(-455 - 1312, -455 + 1312);
47 
48  mFieldLineSeedPoints.clear();
49 
50  for (std::size_t i = 0; i < count; ++i) {
51  mFieldLineSeedPoints.emplace_back(generator_x(rng), generator_y(rng), generator_z(rng));
52  }
53 }

55 template <std::size_t MAX_DIM1_SEGMENTS, std::size_t MAX_DIM2_SEGMENTS, std::size_t MAX_DIM3_SEGMENTS>
56 std::tuple<std::size_t, std::size_t, std::size_t> loadSegments(std::ifstream& file, GPUDisplayMagneticField::SegmentsUniform<MAX_DIM1_SEGMENTS, MAX_DIM2_SEGMENTS, MAX_DIM3_SEGMENTS>& segments)
57 {
58  file.read(reinterpret_cast<char*>(&segments.MinZ), sizeof(segments.MinZ));
59  file.read(reinterpret_cast<char*>(&segments.MaxZ), sizeof(segments.MaxZ));
60  file.read(reinterpret_cast<char*>(&segments.MultiplicativeFactor), sizeof(segments.MultiplicativeFactor));
61 
62  std::int32_t NSegDim1, NSegDim2, NSegDim3;
63 
64  file.read(reinterpret_cast<char*>(&NSegDim1), sizeof(std::int32_t));
65  assert(NSegDim1 <= MAX_DIM1_SEGMENTS);
66 
67  segments.ZSegments = NSegDim1;
68 
69  file.read(reinterpret_cast<char*>(segments.SegDim1), NSegDim1 * sizeof(segments.SegDim1[0]));
70 
71  file.read(reinterpret_cast<char*>(segments.BegSegDim2), NSegDim1 * sizeof(segments.BegSegDim2[0]));
72  file.read(reinterpret_cast<char*>(segments.NSegDim2), NSegDim1 * sizeof(segments.NSegDim2[0]));
73 
74  file.read(reinterpret_cast<char*>(&NSegDim2), sizeof(std::int32_t));
75  assert(NSegDim2 <= MAX_DIM2_SEGMENTS);
76 
77  file.read(reinterpret_cast<char*>(segments.SegDim2), NSegDim2 * sizeof(segments.SegDim2[0]));
78 
79  file.read(reinterpret_cast<char*>(segments.BegSegDim3), NSegDim2 * sizeof(segments.BegSegDim3[0]));
80  file.read(reinterpret_cast<char*>(segments.NSegDim3), NSegDim2 * sizeof(segments.NSegDim3[0]));
81 
82  file.read(reinterpret_cast<char*>(&NSegDim3), sizeof(std::int32_t));
83  assert(NSegDim3 <= MAX_DIM3_SEGMENTS);
84 
85  file.read(reinterpret_cast<char*>(segments.SegDim3), NSegDim3 * sizeof(segments.SegDim3[0]));
86  file.read(reinterpret_cast<char*>(segments.SegID), NSegDim3 * sizeof(segments.SegID[0]));
87 
88  return std::make_tuple(NSegDim1, NSegDim2, NSegDim3);
89 }

91 template <std::size_t DIMENSIONS, std::size_t MAX_PARAMETERIZATIONS, std::size_t MAX_ROWS, std::size_t MAX_COLUMNS, std::size_t MAX_COEFFICIENTS>
92 std::tuple<std::size_t, std::size_t, std::size_t, std::size_t> loadParams(std::ifstream& file, GPUDisplayMagneticField::ParametrizationUniform<MAX_PARAMETERIZATIONS, MAX_ROWS, MAX_COLUMNS, MAX_COEFFICIENTS>& parametrizations)
93 {
94  std::int32_t NParams, NRows, NColumns, NCoefficients;
95  file.read(reinterpret_cast<char*>(&NParams), sizeof(std::int32_t));
96  assert(NParams <= (MAX_PARAMETERIZATIONS / DIMENSIONS));
97 
98  file.read(reinterpret_cast<char*>(parametrizations.BOffsets), DIMENSIONS * NParams * sizeof(parametrizations.BOffsets[0]));
99  file.read(reinterpret_cast<char*>(parametrizations.BScales), DIMENSIONS * NParams * sizeof(parametrizations.BScales[0]));
100  file.read(reinterpret_cast<char*>(parametrizations.BMin), DIMENSIONS * NParams * sizeof(parametrizations.BMin[0]));
101  file.read(reinterpret_cast<char*>(parametrizations.BMax), DIMENSIONS * NParams * sizeof(parametrizations.BMax[0]));
102 
103  file.read(reinterpret_cast<char*>(parametrizations.NRows), DIMENSIONS * NParams * sizeof(parametrizations.NRows[0]));
104  file.read(reinterpret_cast<char*>(parametrizations.ColsAtRowOffset), DIMENSIONS * NParams * sizeof(parametrizations.ColsAtRowOffset[0]));
105  file.read(reinterpret_cast<char*>(parametrizations.CofsAtRowOffset), DIMENSIONS * NParams * sizeof(parametrizations.CofsAtRowOffset[0]));
106 
107  file.read(reinterpret_cast<char*>(&NRows), sizeof(std::int32_t));
108  assert(NRows <= MAX_ROWS);
109 
110  file.read(reinterpret_cast<char*>(parametrizations.NColsAtRow), NRows * sizeof(parametrizations.NColsAtRow[0]));
111  file.read(reinterpret_cast<char*>(parametrizations.CofsAtColOffset), NRows * sizeof(parametrizations.CofsAtColOffset[0]));
112 
113  file.read(reinterpret_cast<char*>(&NColumns), sizeof(std::int32_t));
114  assert(NColumns <= MAX_COLUMNS);
115 
116  file.read(reinterpret_cast<char*>(parametrizations.NCofsAtCol), NColumns * sizeof(parametrizations.NCofsAtCol[0]));
117  file.read(reinterpret_cast<char*>(parametrizations.AtColCoefOffset), NColumns * sizeof(parametrizations.AtColCoefOffset[0]));
118 
119  file.read(reinterpret_cast<char*>(&NCoefficients), sizeof(std::int32_t));
120  assert(NCoefficients <= MAX_COEFFICIENTS);
121 
122  file.read(reinterpret_cast<char*>(parametrizations.Coeffs), NCoefficients * sizeof(parametrizations.Coeffs[0]));
123 
124  return std::make_tuple(NParams, NRows, NColumns, NCoefficients);
125 }

156  mSolCoefficients = SCoeffs;
157 
158  mDipSegDim1 = DipSegDim1;
159  mDipSegDim2 = DipSegDim2;
160  mDipSegDim3 = DipSegDim3;
161 
162  mDipParametrizations = DParams;
163  mDipRows = DRows;

270  auto countArraySizes = [](int numberOfParametrization, auto& getParameter) {
271  int TotalRows = 0, TotalColumns = 0, TotalCoefficients = 0;
272  UShort_t MaxChebyshevOrder = 0;
273 
274  for (auto i = 0; i < numberOfParametrization; ++i) {
275  const auto param = getParameter(i);
276 
277  for (auto j = 0; j < DIMENSIONS; ++j) {
278  const auto calc = param->getChebyshevCalc(j);
279 
280  const auto NRows = calc->getNumberOfRows();
281  TotalRows += NRows;
282 
283  const auto colsAtRow = calc->getNumberOfColumnsAtRow();
284 
285  for (auto row = 0; row < NRows; ++row) {
286  const auto NCols = colsAtRow[row];
287  TotalColumns += NCols;
288 
289  const auto col0 = calc->getColAtRowBg()[row];
290 
291  for (auto col = 0; col < NCols; ++col) {
292  const auto ncoffs = calc->getCoefficientBound2D0()[col0 + col];
293  MaxChebyshevOrder = std::max(MaxChebyshevOrder, ncoffs);
294  }
295  }
296 
297  TotalCoefficients += calc->getNumberOfCoefficients();
298  }
299  }
300 
301  return std::make_tuple(TotalRows, TotalColumns, TotalCoefficients, MaxChebyshevOrder);
302  };
303 
304  const auto [TotalRowsSolenoid, TotalColumnsSolenoid, TotalCoefficientsSolenoid, MaxChebyshevOrderSolenoid] = countArraySizes(numberOfParameterizationSolenoid, getParameterSolenoid);
305  const auto [TotalRowsDipole, TotalColumnsDipole, TotalCoefficientsDipole, MaxChebyshevOrderDipole] = countArraySizes(numberOfParameterizationDipole, getParameterDipole);
306 
307  LOGF(info, "Solenoid Params: %d Rows %d Columns %d Coefficients %d MaxChebOrder: %d", numberOfParameterizationSolenoid, TotalRowsSolenoid, TotalColumnsSolenoid, TotalCoefficientsSolenoid, MaxChebyshevOrderSolenoid);
308  LOGF(info, "Dipole Params: %d Rows %d Columns %d Coefficients %d MaxChebOrder: %d", numberOfParameterizationDipole, TotalRowsDipole, TotalColumnsDipole, TotalCoefficientsDipole, MaxChebyshevOrderDipole);
309 
310  bool spaceOk = true;
311  std::string reason;
312 
313  if (solenoidTableInfo.numberOfDistinctDim1Segments > MAX_SOLENOID_Z_SEGMENTS) {
314  spaceOk = false;
315  reason = "Solenoid Z segment";
316  } else if (solenoidTableInfo.numberOfDistinctDim2Segments > MAX_SOLENOID_P_SEGMENTS) {
317  spaceOk = false;
318  reason = "Solenoid P segment";
319  } else if (solenoidTableInfo.numberOfDistinctDim3Segments > MAX_SOLENOID_R_SEGMENTS) {
320  spaceOk = false;
321  reason = "Solenoid R segment";
322  } else if (dipoleTableInfo.numberOfDistinctDim1Segments > MAX_DIPOLE_Z_SEGMENTS) {
323  spaceOk = false;
324  reason = "Dipole Z Segment";
325  } else if (dipoleTableInfo.numberOfDistinctDim2Segments > MAX_DIPOLE_Y_SEGMENTS) {
326  spaceOk = false;
327  reason = "Dipole Y Segment";
328  } else if (dipoleTableInfo.numberOfDistinctDim3Segments > MAX_DIPOLE_X_SEGMENTS) {
329  spaceOk = false;
330  reason = "Dipole X Segment";
331  } else if (TotalRowsSolenoid > MAX_SOLENOID_ROWS) {
332  spaceOk = false;
333  reason = "Solenoid Rows";
334  } else if (TotalColumnsSolenoid > MAX_SOLENOID_COLUMNS) {
335  spaceOk = false;
336  reason = "Solenoid Columns";
337  } else if (TotalCoefficientsSolenoid > MAX_SOLENOID_COEFFICIENTS) {
338  spaceOk = false;
339  reason = "Solenoid Coefficients";
340  } else if (TotalRowsDipole > MAX_DIPOLE_ROWS) {
341  spaceOk = false;
342  reason = "Dipole Rows";
343  } else if (TotalColumnsDipole > MAX_DIPOLE_COLUMNS) {
344  spaceOk = false;
345  reason = "Dipole Columns";
346  } else if (TotalCoefficientsDipole > MAX_DIPOLE_COEFFICIENTS) {
347  spaceOk = false;
348  reason = "Dipole Coefficients";
349  } else if (std::max(MaxChebyshevOrderSolenoid, MaxChebyshevOrderDipole) > MAX_CHEBYSHEV_ORDER) {
350  spaceOk = false;
351  reason = "Chebyshev Polynomial order";
352  }

355  GPUError("Model has too large %s to fit in the designated uniform", reason);

380  auto initParametrization = [](int numberOfParametrization, auto& getParameter, auto& parametrizationUniform) {
381  int ColsAtRowOffset = 0, CofsAtRowOffset = 0, CofsAtColOffset = 0, Coeffs = 0;
382 
383  for (auto i = 0; i < numberOfParametrization; ++i) {
384  const auto param = getParameter(i);
385  std::copy(param->getBoundaryMappingOffset(), param->getBoundaryMappingOffset() + DIMENSIONS, &parametrizationUniform->BOffsets[i * DIMENSIONS]);
386  std::copy(param->getBoundaryMappingScale(), param->getBoundaryMappingScale() + DIMENSIONS, &parametrizationUniform->BScales[i * DIMENSIONS]);
387  std::copy(param->getBoundMin(), param->getBoundMin() + DIMENSIONS, &parametrizationUniform->BMin[i * DIMENSIONS]);
388  std::copy(param->getBoundMax(), param->getBoundMax() + DIMENSIONS, &parametrizationUniform->BMax[i * DIMENSIONS]);
389 
390  for (auto j = 0; j < DIMENSIONS; ++j) {
391  const auto calc = param->getChebyshevCalc(j);
392 
393  const auto NRows = calc->getNumberOfRows();
394 
395  parametrizationUniform->NRows[i * DIMENSIONS + j] = NRows;
396  parametrizationUniform->ColsAtRowOffset[i * DIMENSIONS + j] = ColsAtRowOffset;
397  parametrizationUniform->CofsAtRowOffset[i * DIMENSIONS + j] = CofsAtRowOffset;
398 
399  const auto colsAtRow = calc->getNumberOfColumnsAtRow();
400 
401  for (auto row = 0; row < NRows; ++row) {
402  const auto NCols = colsAtRow[row];
403  parametrizationUniform->NColsAtRow[ColsAtRowOffset + row] = NCols;
404  parametrizationUniform->CofsAtColOffset[ColsAtRowOffset + row] = CofsAtColOffset;
405 
406  const auto col0 = calc->getColAtRowBg()[row];
407 
408  for (auto col = 0; col < NCols; ++col) {
409  const auto ncoffs = calc->getCoefficientBound2D0()[col0 + col];
410  const auto offset = calc->getCoefficientBound2D1()[col0 + col];
411 
412  parametrizationUniform->NCofsAtCol[CofsAtColOffset + col] = ncoffs;
413  parametrizationUniform->AtColCoefOffset[CofsAtColOffset + col] = offset;
414  }
415  CofsAtColOffset += NCols;
416  }
417 
418  ColsAtRowOffset += NRows;
419  CofsAtRowOffset += calc->getNumberOfCoefficients();
420 
421  std::copy(calc->getCoefficients(), calc->getCoefficients() + calc->getNumberOfCoefficients(), &parametrizationUniform->Coeffs[Coeffs]);
422 
423  Coeffs += calc->getNumberOfCoefficients();
424  }
425  }
426 
427  return std::make_tuple(ColsAtRowOffset, CofsAtColOffset, Coeffs);
428  };
