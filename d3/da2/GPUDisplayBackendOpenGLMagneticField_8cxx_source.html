59 int GPUDisplayBackendOpenGL::InitMagField()
60 {
61  mMagneticField = std::make_unique<GPUDisplayMagneticField>();
62  mMagneticField->generateSeedPoints(mDisplay->cfgL().bFieldLinesCount);
63 
64  CHKERR(mVertexShaderPassthrough = glCreateShader(GL_VERTEX_SHADER));
65  CHKERR(mGeometryShader = glCreateShader(GL_GEOMETRY_SHADER));
66 
67  CHKERR(glShaderSource(mVertexShaderPassthrough, 1, &GPUDisplayShaders::vertexShaderPassthrough, nullptr));
68  CHKERR(glCompileShader(mVertexShaderPassthrough));
69 
70  const auto constantsExpanded = fmt::format(GPUDisplayShaders::fieldModelShaderConstants,
71  fmt::arg("dimensions", GPUDisplayMagneticField::DIMENSIONS),
72  fmt::arg("solZSegs", GPUDisplayMagneticField::MAX_SOLENOID_Z_SEGMENTS),
73  fmt::arg("solPSegs", GPUDisplayMagneticField::MAX_SOLENOID_P_SEGMENTS),
74  fmt::arg("solRSegs", GPUDisplayMagneticField::MAX_SOLENOID_R_SEGMENTS),
75  fmt::arg("solParams", GPUDisplayMagneticField::MAX_SOLENOID_PARAMETERIZATIONS),
76  fmt::arg("solRows", GPUDisplayMagneticField::MAX_SOLENOID_ROWS),
77  fmt::arg("solColumns", GPUDisplayMagneticField::MAX_SOLENOID_COLUMNS),
78  fmt::arg("solCoeffs", GPUDisplayMagneticField::MAX_SOLENOID_COEFFICIENTS),
79  fmt::arg("dipZSegs", GPUDisplayMagneticField::MAX_DIPOLE_Z_SEGMENTS),
80  fmt::arg("dipYSegs", GPUDisplayMagneticField::MAX_DIPOLE_Y_SEGMENTS),
81  fmt::arg("dipXSegs", GPUDisplayMagneticField::MAX_DIPOLE_X_SEGMENTS),
82  fmt::arg("dipParams", GPUDisplayMagneticField::MAX_DIPOLE_PARAMETERIZATIONS),
83  fmt::arg("dipRows", GPUDisplayMagneticField::MAX_DIPOLE_ROWS),
84  fmt::arg("dipColumns", GPUDisplayMagneticField::MAX_DIPOLE_COLUMNS),
85  fmt::arg("dipCoeffs", GPUDisplayMagneticField::MAX_DIPOLE_COEFFICIENTS),
86  fmt::arg("maxChebOrder", GPUDisplayMagneticField::MAX_CHEBYSHEV_ORDER));
87 
88  std::array geomShaderSource = {GPUDisplayShaders::geometryShaderP1, constantsExpanded.c_str(), GPUDisplayShaders::fieldModelShaderCode, GPUDisplayShaders::geometryShaderP2};
89 
90  CHKERR(glShaderSource(mGeometryShader, geomShaderSource.size(), geomShaderSource.data(), nullptr));
91  CHKERR(glCompileShader(mGeometryShader));
92 
93  if (checkShaderStatus(mVertexShaderPassthrough) || checkShaderStatus(mGeometryShader)) {
94  return 1;
95  }
96 
97  CHKERR(mShaderProgramField = glCreateProgram());
98  CHKERR(glAttachShader(mShaderProgramField, mVertexShaderPassthrough));
99  CHKERR(glAttachShader(mShaderProgramField, mGeometryShader));
100  CHKERR(glAttachShader(mShaderProgramField, mFragmentShader));
101  CHKERR(glLinkProgram(mShaderProgramField));
102 
103  if (checkProgramStatus(mShaderProgramField)) {
104  return 1;
105  }
106 
107  const auto ATTRIB_ZERO = 0;
108  const auto BUFFER_IDX = 0;
109 
110  CHKERR(glCreateVertexArrays(1, &VAO_field));
111  CHKERR(glEnableVertexArrayAttrib(VAO_field, ATTRIB_ZERO));
112  CHKERR(glVertexArrayAttribFormat(VAO_field, ATTRIB_ZERO, 3, GL_FLOAT, GL_FALSE, 0));
113 
114  CHKERR(glCreateBuffers(1, &VBO_field));
115  CHKERR(glNamedBufferData(VBO_field, mMagneticField->mFieldLineSeedPoints.size() * sizeof(GPUDisplayMagneticField::vtx), mMagneticField->mFieldLineSeedPoints.data(), GL_STATIC_DRAW));
116 
117  CHKERR(glVertexArrayVertexBuffer(VAO_field, BUFFER_IDX, VBO_field, 0, sizeof(GPUDisplayMagneticField::vtx)));
118 
119  CHKERR(glVertexArrayAttribBinding(VAO_field, ATTRIB_ZERO, BUFFER_IDX));
120 
121  CHKERR(glCreateBuffers(1, &mFieldModelViewBuffer));
122  CHKERR(glNamedBufferData(mFieldModelViewBuffer, sizeof(hmm_mat4), nullptr, GL_STREAM_DRAW));
123 
124  CHKERR(glCreateBuffers(1, &mFieldModelConstantsBuffer));
125  CHKERR(glNamedBufferData(mFieldModelConstantsBuffer, sizeof(GPUDisplayMagneticField::RenderConstantsUniform), mMagneticField->mRenderConstantsUniform.get(), GL_STREAM_DRAW));
126 
127  CHKERR(glCreateBuffers(1, &mSolenoidSegmentsBuffer));
128  CHKERR(glNamedBufferData(mSolenoidSegmentsBuffer, sizeof(GPUDisplayMagneticField::SolenoidSegmentsUniform), mMagneticField->mSolenoidSegments.get(), GL_STREAM_DRAW));
129  CHKERR(glCreateBuffers(1, &mSolenoidParameterizationBuffer));
130  CHKERR(glNamedBufferData(mSolenoidParameterizationBuffer, sizeof(GPUDisplayMagneticField::SolenoidParameterizationUniform), mMagneticField->mSolenoidParameterization.get(), GL_STREAM_DRAW));
131 
132  CHKERR(glCreateBuffers(1, &mDipoleSegmentsBuffer));
133  CHKERR(glNamedBufferData(mDipoleSegmentsBuffer, sizeof(GPUDisplayMagneticField::DipoleSegmentsUniform), mMagneticField->mDipoleSegments.get(), GL_STREAM_DRAW));
134  CHKERR(glCreateBuffers(1, &mDipoleParameterizationBuffer));
135  CHKERR(glNamedBufferData(mDipoleParameterizationBuffer, sizeof(GPUDisplayMagneticField::DipoleParameterizationUniform), mMagneticField->mDipoleParameterization.get(), GL_STREAM_DRAW));
136 
137  return 0;
138 }

140 unsigned int GPUDisplayBackendOpenGL::drawField()
141 {
142  if (!mMagneticField) {
143  return InitMagField(); // next frame will fill MVP matrix
144  }
145 
146  if (mMagneticField->mFieldLineSeedPoints.size() != mDisplay->cfgL().bFieldLinesCount) {
147  mMagneticField->generateSeedPoints(mDisplay->cfgL().bFieldLinesCount);
148  CHKERR(glNamedBufferData(VBO_field, mMagneticField->mFieldLineSeedPoints.size() * sizeof(GPUDisplayMagneticField::vtx), mMagneticField->mFieldLineSeedPoints.data(), GL_STATIC_DRAW));
149  }
150 
151  mMagneticField->mRenderConstantsUniform->StepSize = mDisplay->cfgL().bFieldStepSize;
152  mMagneticField->mRenderConstantsUniform->StepCount = mDisplay->cfgL().bFieldStepCount;
153  CHKERR(glNamedBufferSubData(mFieldModelConstantsBuffer, 0, sizeof(GPUDisplayMagneticField::RenderConstantsUniform), mMagneticField->mRenderConstantsUniform.get()));
154 
155  CHKERR(glBindVertexArray(VAO_field));
156  CHKERR(glUseProgram(mShaderProgramField));
157  CHKERR(glBindBufferBase(GL_UNIFORM_BUFFER, 0, mFieldModelViewBuffer));
158  const std::array<float, 4> drawColor = {1.f, 0.f, 0.f, 1.f};
159  const auto color = glGetUniformLocation(mShaderProgramField, "color");
160  CHKERR(glUniform4fv(color, 1, drawColor.data()));
161 
162  CHKERR(glBindBufferBase(GL_SHADER_STORAGE_BUFFER, 0, mFieldModelConstantsBuffer));
163  CHKERR(glBindBufferBase(GL_SHADER_STORAGE_BUFFER, 1, mSolenoidSegmentsBuffer));
164  CHKERR(glBindBufferBase(GL_SHADER_STORAGE_BUFFER, 2, mDipoleSegmentsBuffer));
165  CHKERR(glBindBufferBase(GL_SHADER_STORAGE_BUFFER, 3, mSolenoidParameterizationBuffer));
166  CHKERR(glBindBufferBase(GL_SHADER_STORAGE_BUFFER, 4, mDipoleParameterizationBuffer));
167 
168  CHKERR(glDrawArrays(GL_POINTS, 0, mMagneticField->mFieldLineSeedPoints.size()));
169 
170  CHKERR(glUseProgram(0));
171  CHKERR(glBindVertexArray(0));
172 
173  return 0;
174 }

182  CHKERR(glDeleteBuffers(1, &mDipoleSegmentsBuffer));

184  CHKERR(glDeleteProgram(mShaderProgramField));
185  CHKERR(glDeleteShader(mGeometryShader));
186  CHKERR(glDeleteShader(mVertexShaderPassthrough));
187  CHKERR(glDeleteBuffers(1, &VBO_field));
188  CHKERR(glDeleteVertexArrays(1, &VAO_field));
189 }
