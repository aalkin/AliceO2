216 int findSolSegment(vec3 pos) {
217  int rid,pid,zid;
218 
219  for(zid=0; zid < sol_segment.ZSegments; zid++) if(pos.z<sol_segment.SegZSol[zid]) break;
220  if(--zid < 0) zid = 0;
221 
222  const int psegBeg = sol_segment.BegSegPSol[zid];
223  for(pid=0; pid<sol_segment.NSegPSol[zid]; pid++) if(pos.y<sol_segment.SegPSol[psegBeg+pid]) break;
224  if(--pid < 0) pid = 0;
225  pid += psegBeg;
226 
227  const int rsegBeg = sol_segment.BegSegRSol[pid];
228  for(rid=0; rid<sol_segment.NSegRSol[pid]; rid++) if(pos.x<sol_segment.SegRSol[rsegBeg+rid]) break;
229  if(--rid < 0) rid = 0;
230  rid += rsegBeg;
231 
232  return sol_segment.SegIDSol[rid];
233 }

270 float cheb1DArray(float x, float arr[MAX_CHEB_ORDER], int ncf) {
271  if(ncf <= 0) return 0.0f;
272 
273  const float x2 = 2*x;
274 
275  vec3 b = vec3(arr[--ncf], 0, 0);
276  --ncf;
277 
278  const vec3 t1 = vec3(1, x2, -1);
279 
280  for (int i=ncf;i>=0;i--) {
281  b.zy = b.yx;
282  b.x = arr[i];
283  b.x = dot(t1, b);
284  }
285 
286  const vec3 t = vec3(1, -x, 0);
287  return dot(t, b);
288 }

352 float Eval3DSol(int segID, int dim, vec3 internal) {
353  const int index = DIMENSIONS*segID;
354  const int n_rows = sol_params.NRows[index+dim];
355  const int cols_at_row_offset = sol_params.ColsAtRowOffset[index+dim];
356  const int coeffs_at_row_offset = sol_params.CofsAtRowOffset[index+dim];
357 
358  for(int row = 0; row < n_rows; row++) {
359  const int n_cols = sol_params.NColsAtRow[cols_at_row_offset+row];
360  const int coeff_at_col_offset = sol_params.CofsAtColOffset[cols_at_row_offset+row];
361 
362  for(int col = 0; col < n_cols; col++) {
363  const int n_coeffs = sol_params.NCofsAtCol[coeff_at_col_offset+col];
364  const int per_col_coeff_offset = sol_params.AtColCoefOffset[coeff_at_col_offset+col];
365 
366  const int coeffs_offset = coeffs_at_row_offset + per_col_coeff_offset;
367 
368  tmpCfs1[col] = cheb1DParamsSol(internal.z, coeffs_offset,n_coeffs);
369  }
370  tmpCfs0[row] = cheb1DArray(internal.y, tmpCfs1, n_cols);
371  }
372 
373  return cheb1DArray(internal.x, tmpCfs0, n_rows);
374 }

410 vec3 CyltoCartCylB(vec3 rphiz, vec3 brphiz) {
411  const float btr = length(brphiz.xy);
412  const float psiPLUSphi = atan(brphiz.y, brphiz.x) + rphiz.y;

414  return vec3(btr*cos(psiPLUSphi), btr*sin(psiPLUSphi), brphiz.z);

421 vec3 SolDipField(vec3 pos) {
422  if(pos.z > sol_segment.MinZ) {
423  const vec3 rphiz = CarttoCyl(pos);
424  const int segID = findSolSegment(rphiz);
425  if(segID >=0 && IsInsideSol(segID, rphiz)) {
426  const vec3 brphiz = EvalSol(segID, rphiz);
427  return CyltoCartCylB(rphiz, brphiz) * sol_segment.MultiplicativeFactor;
428  }
429  }
430 
431  const int segID = findDipSegment(pos);
432  if(segID >= 0 && IsInsideDip(segID, pos)) {
433  return EvalDip(segID, pos) * dip_segment.MultiplicativeFactor;
434  }
435 
436  return vec3(0);
437 }
438 
439 const float MinZ = dip_segment.MinZ;
440 const float MaxZ = sol_segment.MaxZ;
441 
442 vec3 Field(vec3 pos) {
443  if(pos.z > MinZ && pos.z < MaxZ) {
444  return SolDipField(pos);
445  }
446  return vec3(0);
