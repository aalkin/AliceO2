40 int FT0DCSProcessor::process(const gsl::span<const DPCOM> dps)
41 {
42  // first we check which DPs are missing - if some are, it means that
43  // the delta map was sent
44 
45  if (mVerbose) {
46  LOG(info) << "\n\nProcessing new DCS DP map\n-------------------------";
47  }
48 
49  if (false) {
50  std::unordered_map<DPID, DPVAL> mapin;
51  for (auto& it : dps) {
52  mapin[it.id] = it.data;
53  }
54  for (auto& it : mPids) {
55  const auto& el = mapin.find(it.first);
56  if (el == mapin.end()) {
57  LOG(debug) << "DP " << it.first << " not found in map";
58  } else {
59  LOG(debug) << "DP " << it.first << " found in map";
60  }
61  }
62  }
63 
64  // now we process all DPs, one by one
65  for (const auto& it : dps) {
66  // we process only the DPs defined in the configuration
67  const auto& el = mPids.find(it.id);
68  if (el == mPids.end()) {
69  LOG(info) << "DP " << it.id << " not found in FT0DCSProcessor, we will not process it";
70  continue;
71  }
72  processDP(it);
73  mPids[it.id] = true;
74  }
75 
76  return 0;
77 }

79 int FT0DCSProcessor::processDP(const DPCOM& dpcom)
80 {
81  // processing a single DP
82 
83  const auto& dpid = dpcom.id;
84  const auto& type = dpid.get_type();
85  const auto& val = dpcom.data;
86  if (mVerbose) {
87  if (type == DPVAL_FLOAT) {
88  LOG(info) << "Processing DP = " << dpcom << " (epoch " << val.get_epoch_time() << "), with value = " << o2::dcs::getValue<float>(dpcom);
89  } else if (type == DPVAL_INT) {
90  LOG(info) << "Processing DP = " << dpcom << " (epoch " << val.get_epoch_time() << "), with value = " << o2::dcs::getValue<int32_t>(dpcom);
91  }
92  }
93  auto flags = val.get_flags();
94  if (processFlags(flags, dpid.get_alias()) == 0) {
95  // Store all DP values
96  if (mFT0DCS[dpid].values.empty() || val.get_epoch_time() > mFT0DCS[dpid].values.back().first) {
97  converter.raw_data = val.payload_pt1;
98  if (type == DPVAL_FLOAT) {
99  mFT0DCS[dpid].add(val.get_epoch_time(), lround(converter.float_value * 1000)); // store as nA
100  } else if (type == DPVAL_INT) {
101  mFT0DCS[dpid].add(val.get_epoch_time(), converter.int_value);
102  }
103  }
104  }
105  return 0;
106 }
