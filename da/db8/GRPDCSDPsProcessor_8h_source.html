59 struct MagFieldHelper {
60  int isSet = 0;
61  float curL3 = 0.;
62  float curDip = 0.;
63  bool negL3 = 0;
64  bool negDip = 0;
65  bool updated = false;
66  bool verbose = false;
67 
68  void updateCurL3(float v)
69  {
70  if (!(isSet & 0x1) || std::abs(v - curL3) > 0.1) {
71  if (verbose) {
72  LOG(info) << "L3 current will be updated from " << curL3 << " to " << v;
73  }
74  curL3 = v;
75  isSet |= 0x1;
76  updated = true;
77  }
78  }
79  void updateCurDip(float v)
80  {
81  if (!(isSet & 0x2) || std::abs(v - curDip) > 0.1) {
82  if (verbose) {
83  LOG(info) << "Dipole current will be updated from " << curDip << " to " << v;
84  }
85  curDip = v;
86  isSet |= 0x2;
87  updated = true;
88  }
89  }
90  void updateSignL3(bool v)
91  {
92  if (!(isSet & 0x4) || v != negL3) {
93  if (verbose) {
94  LOG(info) << "L3 polarity will be updated from " << negL3 << " to " << v;
95  }
96  negL3 = v;
97  isSet |= 0x4;
98  updated = true;
99  }
100  }
101  void updateSignDip(bool v)
102  {
103  if (!(isSet & 0x8) || v != negDip) {
104  if (verbose) {
105  LOG(info) << "Dipole polarity will be updated from " << negDip << " to " << v;
106  }
107  negDip = v;
108  isSet |= 0x8;
109  updated = true;
110  }
111  }
