40 void DEDigitizer::process(const Hit& hit, int evID, int srcID)
41 {
42  MCCompLabel label(hit.GetTrackID(), evID, srcID);
43 
44  // convert energy to charge
45  auto charge = mResponse.etocharge(hit.GetEnergyLoss());
46  auto chargeCorr = mResponse.chargeCorr();
47  auto chargeBending = chargeCorr * charge;
48  auto chargeNonBending = charge / chargeCorr;
49 
50  // local position of the charge distribution
51  Point3D<float> pos(hit.GetX(), hit.GetY(), hit.GetZ());
52  Point3D<float> lpos;
53  mTransformation.MasterToLocal(pos, lpos);
54  auto localX = mResponse.getAnod(lpos.X());
55  auto localY = lpos.Y();
56 
57  // borders of charge integration area
58  auto dxy = mResponse.getSigmaIntegration() * mResponse.getChargeSpread();
59  auto xMin = localX - dxy;
60  auto xMax = localX + dxy;
61  auto yMin = localY - dxy;
62  auto yMax = localY + dxy;
63 
64  // loop over all pads within the defined bounding box to compute the charge on each of them
65  mSegmentation.forEachPadInArea(xMin, yMin, xMax, yMax, [&](int padid) {
66  auto dx = mSegmentation.padSizeX(padid) * 0.5;
67  auto dy = mSegmentation.padSizeY(padid) * 0.5;
68  auto xPad = mSegmentation.padPositionX(padid) - localX;
69  auto yPad = mSegmentation.padPositionY(padid) - localY;
70  auto q = mResponse.chargePadfraction(xPad - dx, xPad + dx, yPad - dy, yPad + dy);
71  if (mResponse.isAboveThreshold(q)) {
72  q *= mSegmentation.isBendingPad(padid) ? chargeBending : chargeNonBending;
73  if (q > 0.f) {
74  mCharges[padid] += q;
75  mLabels[padid].emplace_back(label);
76  }
77  }
78  });
79 }

81 void DEDigitizer::addNoise(float noiseProba)
82 {
83  std::random_device rd;
84  std::mt19937 mt(rd());
85 
86  float mean = noiseProba * mSegmentation.nofPads();
87  float sigma = mean / std::sqrt(mean);
88 
89  std::normal_distribution<float> gaus(mean, sigma);
90 
91  int nofNoisyPads = std::ceil(gaus(mt));
92 
93  std::uniform_int_distribution<int> ids(0, mSegmentation.nofPads() - 1);
94 
95  float chargeNoise = 1.2;
96  // FIXME: draw this also from some distribution (according to
97  // some parameters in DigitizerParam)
98  for (auto i = 0; i < nofNoisyPads; i++) {
99  auto padid = ids(mt);
100  mCharges[padid] += chargeNoise;
101  mLabels[padid].emplace_back(true);
102  }
103 }

105 void DEDigitizer::extractDigitsAndLabels(std::vector<Digit>& digits,
106  o2::dataformats::MCTruthContainer<o2::MCCompLabel>& labels)
107 {
108  int dataindex = labels.getIndexedSize();
109  for (auto padid = 0; padid < mCharges.size(); ++padid) {
110  auto q = mCharges[padid];
111  if (q <= 0.f) {
112  continue;
113  }
114  uint32_t adc = std::round(q);
115  // FIXME: we should handle charge saturation here
116  if (adc >= DigitizerParam::Instance().minADC) {
117  auto time = mIR.differenceInBC(o2::raw::HBFUtils::Instance().orbitFirst);
118  auto nSamples = mResponse.nSamples(adc);
119  digits.emplace_back(mDeId, padid, adc, time, nSamples);
120  for (auto element : mLabels[padid]) {
121  labels.addElement(dataindex, element);
122  }
123  ++dataindex;
124  }
125  }
126 }
