69 bool DataReaderTask::isTimeFrameEmpty(ProcessingContext& pc)
70 {
71  constexpr auto origin = header::gDataOriginTRD;
72  o2::framework::InputSpec dummy{"dummy", framework::ConcreteDataMatcher{origin, header::gDataDescriptionRawData, 0xDEADBEEF}};
73  // if we see requested data type input with 0xDEADBEEF subspec and 0 payload.
74  // frame detected we have no data and send this instead
75  // send empty output so as to not block workflow
76  static size_t contDeadBeef = 0; // number of times 0xDEADBEEF was seen continuously
77  for (const auto& ref : o2::framework::InputRecordWalker(pc.inputs(), {dummy})) {
78  const auto dh = o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
79  auto payloadSize = o2::framework::DataRefUtils::getPayloadSize(ref);
80  if (payloadSize == 0) {
81  auto maxWarn = o2::conf::VerbosityConfig::Instance().maxWarnDeadBeef;
82  if (++contDeadBeef <= maxWarn) {
83  LOGP(alarm, "Found input [{}/{}/{:#x}] TF#{} 1st_orbit:{} Payload {} : assuming no payload for all links in this TF{}",
84  dh->dataOrigin.str, dh->dataDescription.str, dh->subSpecification, dh->tfCounter, dh->firstTForbit, payloadSize,
85  contDeadBeef == maxWarn ? fmt::format(". {} such inputs in row received, stopping reporting", contDeadBeef) : "");
86  }
87  return true;
88  }
89  }
90  contDeadBeef = 0; // if good data, reset the counter
91  return false;
92 }

94 void DataReaderTask::run(ProcessingContext& pc)
95 {
96  //NB this is run per time frame on the epn.
97  LOG(info) << "TRD Translator Task run";
98  auto dataReadStart = std::chrono::high_resolution_clock::now();
99 
100  if (isTimeFrameEmpty(pc)) {
101  sendData(pc, true); //send the empty tf data.
102  return;
103  }
104  uint64_t total1 = 0, total2 = 0;
105 
106  std::vector<InputSpec> sel{InputSpec{"filter", ConcreteDataTypeMatcher{"TRD", "RAWDATA"}}};
107  uint64_t tfCount = 0;
108  for (auto& ref : InputRecordWalker(pc.inputs(), sel)) {
109  auto inputprocessingstart = std::chrono::high_resolution_clock::now(); // measure total processing time
110  const auto* dh = DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
111  tfCount = dh->tfCounter;
112  const char* payloadIn = ref.payload;
113  auto payloadInSize = DataRefUtils::getPayloadSize(ref);
114  if (mHeaderVerbose) {
115  LOGP(info, "Found input [{}/{}/{:#x}] TF#{} 1st_orbit:{} Payload {} : ",
116  dh->dataOrigin.str, dh->dataDescription.str, dh->subSpecification, dh->tfCounter, dh->firstTForbit, payloadInSize);
117  }
118  if (!mCompressedData) { //we have raw data coming in from flp
119  if (mVerbose) {
120  LOG(info) << " parsing non compressed data in the data reader task with a payload of " << payloadInSize << " payload size";
121  }
122  // LOG(info) << "start of data is at ref.payload=0x"<< std::hex << " payloadSize:0x" << payloadInSize <<" dh->headerSize:0x" <<dh->headerSize;
123  total1 += payloadInSize;
124  total2 += dh->headerSize;
125  // LOG(info) << "start of data is at ref.payload=0x"<< std::hex << " total1:0x" << total1 <<" total2:0x" <<total2;
126  mReader.setDataBuffer(payloadIn);
127  mReader.setDataBufferSize(payloadInSize);
128  mReader.configure(mTrackletHCHeaderState, mHalfChamberWords, mHalfChamberMajor, mOptions);
129  //mReader.setStats(&mTimeFrameStats);
130  mReader.run();
131  mWordsRead += mReader.getWordsRead();
132  mWordsRejected += mReader.getWordsRejected();
133  if (mVerbose) {
134  LOG(info) << "relevant vectors to read : " << mReader.sumTrackletsFound() << " tracklets and " << mReader.sumDigitsFound() << " compressed digits";
135  }
136  } else { // we have compressed data coming in from flp.
137  mCompressedReader.setDataBuffer(payloadIn);
138  mCompressedReader.setDataBufferSize(payloadInSize);
139  mCompressedReader.configure(mOptions);
140  mCompressedReader.run();
141  }
142  }
143 
144  sendData(pc, false);
145  std::chrono::duration<double, std::milli> dataReadTime = std::chrono::high_resolution_clock::now() - dataReadStart;
146  LOGP(info, "Digits: {}, Tracklets: {}, DataRead in: {:.3f} MB, Rejected: {:.3f} MB for TF {} in {} ms",
147  mReader.getDigitsFound(), mReader.getTrackletsFound(), (float)mWordsRead * 4 / 1024.0 / 1024.0, (float)mWordsRejected * 4 / 1024.0 / 1024.0, tfCount,
148  std::chrono::duration_cast<std::chrono::milliseconds>(dataReadTime).count());
149 }
