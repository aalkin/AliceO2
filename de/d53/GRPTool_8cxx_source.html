109 void setROMode(std::string const& filename, std::vector<std::string> const& continuous,
110  std::vector<std::string> const& triggered, bool clear = false)
111 {
112  using o2::detectors::DetID;
113 
114  if (filename.size() == 0) {
115  std::cout << "no filename given\n";
116  return;
117  }
118  using GRPECSObject = o2::parameters::GRPECSObject;
119  const std::string grpName{o2::base::NameConf::CCDBOBJECT};
120  TFile flGRP(filename.c_str(), "update");
121  if (flGRP.IsZombie()) {
122  LOG(error) << "Failed to open GRPECS file " << filename << " in update mode ";
123  return;
124  }
125  std::unique_ptr<GRPECSObject> grp(static_cast<GRPECSObject*>(flGRP.GetObjectChecked(grpName.c_str(), GRPECSObject::Class())));
126  if (grp.get()) {
127  // clear complete state (continuous state) first of all when asked
128  if (clear) {
129  for (auto id = DetID::First; id <= DetID::Last; ++id) {
130  if (grp->isDetReadOut(id)) {
131  grp->remDetContinuousReadOut(id);
132  }
133  }
134  }
135 
136  //
137  for (auto& detstr : continuous) {
138  // convert to detID
139  o2::detectors::DetID id(detstr.c_str());
140  if (grp->isDetReadOut(id)) {
141  grp->addDetContinuousReadOut(id);
142  LOG(info) << "Setting det " << detstr << " to continuous RO mode";
143  }
144  }
145  //
146  for (auto& detstr : triggered) {
147  // convert to detID
148  o2::detectors::DetID id(detstr.c_str());
149  if (grp->isDetReadOut(id)) {
150  grp->addDetTrigger(id);
151  LOG(info) << "Setting det " << detstr << " to trigger CTP";
152  }
153  }
154  grp->print();
155  flGRP.WriteObjectAny(grp.get(), grp->Class(), grpName.c_str());
156  }
157  flGRP.Close();
158 }

160 // copies a file idendified by filename to a CCDB snapshot starting under path
161 // and with the CCDBpath hierarchy
162 bool publish(std::string const& filename, std::string const& path, std::string CCDBpath)
163 {
164  if (!std::filesystem::exists(filename)) {
165  LOG(error) << "Input file " << filename << "does not exist\n";
166  return false;
167  }
168 
169  std::string targetdir = path + CCDBpath;
170  try {
171  o2::utils::createDirectoriesIfAbsent(targetdir);
172  } catch (std::exception e) {
173  LOGP(error, fmt::format("Could not create local snapshot cache directory {}, reason: {}", targetdir, e.what()));
174  return false;
175  }
176 
177  auto targetfile = std::filesystem::path(targetdir + "/snapshot.root");
178  auto opts = std::filesystem::copy_options::overwrite_existing;
179  std::filesystem::copy_file(filename, targetfile, opts);
180  if (std::filesystem::exists(targetfile)) {
181  LOG(info) << "file " << filename << " copied/published to " << targetfile;
182  }
183  return true;
184 }

209 // creates a set of basic GRP files (for simulation)
210 bool create_GRPs(Options const& opts)
211 {
212  // some code duplication from o2-sim --> remove it
213 
214  uint64_t runStart = -1; // used in multiple GRPs
215 
216  // GRPECS
217  {
218  LOG(info) << " --- creating GRP ECS -----";
219  o2::parameters::GRPECSObject grp;
220  grp.setRun(opts.run);
221  // if
222  auto& ccdbmgr = o2::ccdb::BasicCCDBManager::instance();
223  auto soreor = ccdbmgr.getRunDuration(opts.run);
224  runStart = soreor.first;
225  grp.setTimeStart(runStart);
226  grp.setTimeEnd(runStart + 3600000);
227  grp.setNHBFPerTF(opts.orbitsPerTF);
228  std::vector<std::string> modules{};
229  o2::conf::SimConfig::determineActiveModules(opts.readout, std::vector<std::string>(), modules);
230  std::vector<std::string> readout{};
231  o2::conf::SimConfig::determineReadoutDetectors(modules, std::vector<std::string>(), opts.skipreadout, readout);
232  for (auto& detstr : readout) {
233  o2::detectors::DetID id(detstr.c_str());
234  grp.addDetReadOut(id);
235  // set default RO modes
236  if (!o2::parameters::GRPECSObject::alwaysTriggeredRO(id)) {
237  grp.addDetContinuousReadOut(id);
238  }
239  }
240  grp.setIsMC(true);
241  grp.setRunType(o2::parameters::GRPECSObject::PHYSICS);
242  // grp.setDataPeriod("mc"); // decide what to put here
243  std::string grpfilename = o2::base::NameConf::getGRPECSFileName(opts.outprefix);
244  TFile grpF(grpfilename.c_str(), "recreate");
245  grpF.WriteObjectAny(&grp, grp.Class(), o2::base::NameConf::CCDBOBJECT.data());
246  grpF.Close();
247  if (opts.print) {
248  grp.print();
249  }
250  if (opts.publishto.size() > 0) {
251  publish(grpfilename, opts.publishto, "/GLO/Config/GRPECS");
252  }
253  }
254 
255  // GRPMagField
256  {
257  LOG(info) << " --- creating magfield GRP -----";
258  o2::parameters::GRPMagField grp;
259  // parse the wanted field value
260  int fieldvalue = 0;
261  o2::conf::SimFieldMode fieldmode;
262  auto ok = o2::conf::SimConfig::parseFieldString(opts.fieldstring, fieldvalue, fieldmode);
263  if (!ok) {
264  LOG(error) << "Error parsing field string " << opts.fieldstring;
265  return false;
266  }
267 
268  // let's not create an actual mag field object for this
269  // we only need to lookup the currents from the possible
270  // values of mag field
271  // +-2,+-5,0 and uniform
272 
273  const std::unordered_map<int, std::pair<int, int>> field_to_current = {{2, {12000, 6000}},
274  {5, {30000, 6000}},
275  {-2, {-12000, -6000}},
276  {-5, {-30000, -6000}},
277  {0, {0, 0}}};
278 
279  auto currents_iter = field_to_current.find(fieldvalue);
280  if (currents_iter == field_to_current.end()) {
281  LOG(error) << " Could not lookup currents for fieldvalue " << fieldvalue;
282  return false;
283  }
284 
285  o2::units::Current_t currDip = (*currents_iter).second.first;
286  o2::units::Current_t currL3 = (*currents_iter).second.second;
287  grp.setL3Current(currL3);
288  grp.setDipoleCurrent(currDip);
289  grp.setFieldUniformity(fieldmode == o2::conf::SimFieldMode::kUniform);
290  if (opts.print) {
291  grp.print();
292  }
293  std::string grpfilename = o2::base::NameConf::getGRPMagFieldFileName(opts.outprefix);
294  TFile grpF(grpfilename.c_str(), "recreate");
295  grpF.WriteObjectAny(&grp, grp.Class(), o2::base::NameConf::CCDBOBJECT.data());
296  grpF.Close();
297  if (opts.publishto.size() > 0) {
298  publish(grpfilename, opts.publishto, "/GLO/Config/GRPMagField");
299  }
300  }
301 
302  // GRPLHCIF --> complete it later
303  {
304  LOG(info) << " --- creating GRP LHCIF -----";
305  o2::parameters::GRPLHCIFData grp;
306  // eventually we need to set the beam info from the generator, at the moment put some plausible values
307  grp.setFillNumberWithTime(runStart, 0); // RS FIXME
308  grp.setInjectionSchemeWithTime(runStart, ""); // RS FIXME
309  grp.setBeamEnergyPerZWithTime(runStart, 6.8e3); // RS FIXME
310  grp.setAtomicNumberB1WithTime(runStart, 1.); // RS FIXME
311  grp.setAtomicNumberB2WithTime(runStart, 1.); // RS FIXME
312  grp.setCrossingAngleWithTime(runStart, 0.); // RS FIXME
313  grp.setBeamAZ();
314 
315  // set the BC pattern if necessary
316  if (opts.bcPatternFile.size() > 0) {
317  // load bunch filling from the file
318  auto* bc = o2::BunchFilling::loadFrom(opts.bcPatternFile);
319  if (!bc) {
320  LOG(fatal) << "Failed to load bunch filling from " << opts.bcPatternFile;
321  }
322  grp.setBunchFillingWithTime(grp.getBeamEnergyPerZTime(), *bc); // borrow the time from the existing entry
323  delete bc;
324  }
325  std::string grpfilename = o2::base::NameConf::getGRPLHCIFFileName(opts.outprefix);
326  if (opts.print) {
327  grp.print();
328  }
329  TFile grpF(grpfilename.c_str(), "recreate");
330  grpF.WriteObjectAny(&grp, grp.Class(), o2::base::NameConf::CCDBOBJECT.data());
331  grpF.Close();
332  if (opts.publishto.size() > 0) {
333  publish(grpfilename, opts.publishto, "/GLO/Config/GRPLHCIF");
334  }
335  }
336 
337  return true;
338 }

340 void perform_Command(Options const& opts)
341 {
342  switch (opts.command) {
343  case GRPCommand::kCREATE: {
344  create_GRPs(opts);
345  break;
346  }
347  case GRPCommand::kANCHOR: {
348  anchor_GRPs(opts);
349  break;
350  }
351  case GRPCommand::kPRINTECS: {
352  printGRPECS(opts.grpfilename);
353  break;
354  }
355  case GRPCommand::kPRINTMAG: {
356  printGRPMAG(opts.grpfilename);
357  break;
358  }
359  case GRPCommand::kPRINTLHC: {
360  printGRPLHC(opts.grpfilename);
361  break;
362  }
363  case GRPCommand::kSETROMODE: {
364  setROMode(opts.grpfilename, opts.continuous, opts.triggered, opts.clearRO);
365  break;
366  }
367  default: {
368  }
369  }
370 }

372 bool parseOptions(int argc, char* argv[], Options& optvalues)
373 {
374  namespace bpo = boost::program_options;
375  bpo::options_description global("Global options");
376  global.add_options()("command", bpo::value<std::string>(), "command to execute")("subargs", bpo::value<std::vector<std::string>>(), "Arguments for command");
377  global.add_options()("help,h", "Produce help message.");
378 
379  bpo::positional_options_description pos;
380  pos.add("command", 1).add("subargs", -1);
381 
382  bpo::variables_map vm;
383  bpo::parsed_options parsed{nullptr};
384  try {
385  parsed = bpo::command_line_parser(argc, argv).options(global).positional(pos).allow_unregistered().run();
386 
387  bpo::store(parsed, vm);
388 
389  // help
390  if (vm.count("help") > 0 && vm.count("command") == 0) {
391  print_globalHelp(argc, argv);
392  return false;
393  }
394  } catch (const bpo::error& e) {
395  std::cerr << e.what() << "\n\n";
396  std::cerr << "Error parsing global options; Available options:\n";
397  std::cerr << global << std::endl;
398  return false;
399  }
400 
401  auto subparse = [&parsed](auto& desc, auto& vm, std::string const& command_name) {
402  try {
403  // Collect all the unrecognized options from the first pass. This will include the
404  // (positional) command name, so we need to erase that.
405  std::vector<std::string> opts = bpo::collect_unrecognized(parsed.options, bpo::include_positional);
406  // opts.erase(opts.begin());
407 
408  // Parse again... and store to vm
409  bpo::store(bpo::command_line_parser(opts).options(desc).run(), vm);
410  bpo::notify(vm);
411 
412  if (vm.count("help")) {
413  std::cout << desc << std::endl;
414  return false;
415  }
416  } catch (const bpo::error& e) {
417  std::cerr << e.what() << "\n\n";
418  std::cerr << "Error parsing options for " << command_name << " Available options:\n";
419  std::cerr << desc << std::endl;
420  return false;
421  }
422  return true;
423  };
424 
425  std::string cmd = vm["command"].as<std::string>();
426 
427  if (cmd == "anchorGRPs") {
428  optvalues.command = GRPCommand::kANCHOR;
429  // ls command has the following options:
430  bpo::options_description desc("anchor GRP options");
431 
432  // ls command has the following options:
433  desc.add_options()("run", bpo::value<int>(&optvalues.run)->default_value(-1), "Run number");
434  desc.add_options()("print", "print resulting GRPs");
435  desc.add_options()("publishto", bpo::value<std::string>(&optvalues.publishto)->default_value("GRP"), "Base path under which GRP objects should be published on disc. This path can serve as lookup for CCDB queries of the GRP objects.");
436  if (!subparse(desc, vm, "anchorGRPs")) {
437  return false;
438  }
439  if (vm.count("print") > 0) {
440  optvalues.print = true;
441  }
442  } else if (cmd == "createGRPs") {
443  optvalues.command = GRPCommand::kCREATE;
444 
445  // ls command has the following options:
446  bpo::options_description desc("create options");
447  desc.add_options()("readoutDets", bpo::value<std::vector<std::string>>(&optvalues.readout)->multitoken()->default_value(std::vector<std::string>({"all"}), "all Run3 detectors"), "Detector list to be readout/active");
448  desc.add_options()("skipReadout", bpo::value<std::vector<std::string>>(&optvalues.skipreadout)->multitoken()->default_value(std::vector<std::string>(), "nothing skipped"), "list of inactive detectors (precendence over --readout)");
449  desc.add_options()("run", bpo::value<int>(&optvalues.run)->default_value(-1), "Run number");
450  desc.add_options()("field", bpo::value<std::string>(&optvalues.fieldstring)->default_value("-5"), "L3 field rounded to kGauss, allowed values +-2,+-5 and 0; +-<intKGaus>U for uniform field");
451  desc.add_options()("outprefix,o", bpo::value<std::string>(&optvalues.outprefix)->default_value("o2sim"), "Prefix for GRP output files");
452  desc.add_options()("bcPatternFile", bpo::value<std::string>(&optvalues.bcPatternFile)->default_value(""), "Interacting BC pattern file (e.g. from CreateBCPattern.C)");
453  desc.add_options()("print", "print resulting GRPs");
454  desc.add_options()("publishto", bpo::value<std::string>(&optvalues.publishto)->default_value(""), "Base path under which GRP objects should be published on disc. This path can serve as lookup for CCDB queries of the GRP objects.");
455  if (!subparse(desc, vm, "createGRPs")) {
456  return false;
457  }
458  if (vm.count("print") > 0) {
459  optvalues.print = true;
460  }
461  } else if (cmd == "setROMode") {
462  // set/modify the ROMode
463  optvalues.command = GRPCommand::kSETROMODE;
464  bpo::options_description desc("setting detector readout modes");
465  desc.add_options()("file,f", bpo::value<std::string>(&optvalues.grpfilename)->default_value("o2sim_grpecs.root"), "Path to GRPECS file");
466  desc.add_options()("continuousRO", bpo::value<std::vector<std::string>>(&optvalues.continuous)->multitoken()->default_value(std::vector<std::string>({"all"}), "all active detectors"), "List of detectors to set to continuous mode");
467  desc.add_options()("triggerCTP", bpo::value<std::vector<std::string>>(&optvalues.triggered)->multitoken()->default_value(std::vector<std::string>({""}), "none"), "List of detectors to trigger CTP");
468  desc.add_options()("clear", "clears all RO modes (prio to applying other options)");
469  if (!subparse(desc, vm, "setROMode")) {
470  return false;
471  }
472  if (vm.count("clear") > 0) {
473  optvalues.clearRO = true;
474  }
475  } else if (cmd == "print_GRPECS") {
476  optvalues.command = GRPCommand::kPRINTECS;
477  // print the GRP
478  bpo::options_description desc("print options");
479  desc.add_options()("file,f", bpo::value<std::string>(&optvalues.grpfilename), "Path to GRP file");
480  if (!subparse(desc, vm, "print_GRPECS")) {
481  return false;
482  }
483  } else if (cmd == "print_GRPLHC") {
484  optvalues.command = GRPCommand::kPRINTLHC;
485  // print the GRP
486  bpo::options_description desc("print options");
487  desc.add_options()("file,f", bpo::value<std::string>(&optvalues.grpfilename), "Path to GRP file");
488  if (!subparse(desc, vm, "print_GRPECS")) {
489  return false;
490  }
491  } else if (cmd == "print_GRPMAG") {
492  optvalues.command = GRPCommand::kPRINTMAG;
493  // print the GRP
494  bpo::options_description desc("print options");
495  desc.add_options()("file,f", bpo::value<std::string>(&optvalues.grpfilename), "Path to GRP file");
496  if (!subparse(desc, vm, "print_GRPECS")) {
497  return false;
498  }
499  } else {
500  std::cerr << "Error: Unknown command " << cmd << std::endl;
501  return false;
502  }
503 
504  return true;
505 }
