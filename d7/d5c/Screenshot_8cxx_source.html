30 bool Screenshot::CopyImage(TASImage* dst, TASImage* src, Int_t x_dst, Int_t y_dst, Int_t x_src, Int_t y_src,
31  UInt_t w_src, UInt_t h_src)
32 {
33  if (!dst) {
34  return false;
35  }
36  if (!src) {
37  return false;
38  }
39 
40  int x = 0;
41  int y = 0;
42  int idx_src = 0;
43  int idx_dst = 0;
44  x_src = x_src < 0 ? 0 : x_src;
45  y_src = y_src < 0 ? 0 : y_src;
46 
47  if ((x_src >= (int)src->GetWidth()) || (y_src >= (int)src->GetHeight())) {
48  return false;
49  }
50 
51  w_src = x_src + w_src > src->GetWidth() ? src->GetWidth() - x_src : w_src;
52  h_src = y_src + h_src > src->GetHeight() ? src->GetHeight() - y_src : h_src;
53  UInt_t yy = (y_src + y) * src->GetWidth();
54 
55  src->BeginPaint(false);
56  dst->BeginPaint(false);
57 
58  UInt_t* dst_image_array = dst->GetArgbArray();
59  UInt_t* src_image_array = src->GetArgbArray();
60 
61  if (!dst_image_array || !src_image_array) {
62  return false;
63  }
64 
65  for (y = 0; y < (int)h_src; y++) {
66  for (x = 0; x < (int)w_src; x++) {
67 
68  idx_src = yy + x + x_src;
69  idx_dst = (y_dst + y) * dst->GetWidth() + x + x_dst;
70 
71  if ((x + x_dst < 0) || (y_dst + y < 0) ||
72  (x + x_dst >= (int)dst->GetWidth()) || (y + y_dst >= (int)dst->GetHeight())) {
73  continue;
74  }
75 
76  dst_image_array[idx_dst] = src_image_array[idx_src];
77  }
78  yy += src->GetWidth();
79  }
80 
81  return true;

84 TASImage* Screenshot::ScaleImage(TASImage* image, UInt_t desiredWidth, UInt_t desiredHeight)
85 {
86  if (!image) {
87  return nullptr;
88  }
89  if (desiredWidth == 0 || desiredHeight == 0) {
90  return nullptr;
91  }
92 
93  const char* backgroundColor = "#000000";
94 
95  UInt_t scaleWidth = desiredWidth;
96  UInt_t scaleHeight = desiredHeight;
97  UInt_t offsetWidth = 0;
98  UInt_t offsetHeight = 0;
99 
100  float aspectRatio = (float)image->GetWidth() / (float)image->GetHeight();
101 
102  if (desiredWidth >= aspectRatio * desiredHeight) {
103  scaleWidth = (UInt_t)(aspectRatio * desiredHeight);
104  offsetWidth = (desiredWidth - scaleWidth) / 2.0f;
105  } else {
106  scaleHeight = (UInt_t)((1.0f / aspectRatio) * desiredWidth);
107  offsetHeight = (desiredHeight - scaleHeight) / 2.0f;
108  }
109 
110  TASImage* scaledImage = new TASImage(desiredWidth, desiredHeight);
111  scaledImage->FillRectangle(backgroundColor, 0, 0, desiredWidth, desiredHeight);
112 
113  image->Scale(scaleWidth, scaleHeight);
114 
115  CopyImage(scaledImage, image, offsetWidth, offsetHeight, 0, 0, scaleWidth, scaleHeight);
116 
117  return scaledImage;
118 }
