53 float Response::etocharge(float edepos) const
54 {
55  int nel = int(edepos * 1.e9 / 27.4);
56  if (nel == 0) {
57  nel = 1;
58  }
59  float charge = 0.f;
60  for (int i = 1; i <= nel; i++) {
61  float arg = 0.f;
62  do {
63  arg = gRandom->Rndm();
64  } while (!arg);
65  charge -= mChargeSlope * TMath::Log(arg);
66  }
67  return charge;
68 }

85 uint16_t Response::nSamples(uint32_t adc) const
86 {
87  // the main purpose is to the pass the background rejection and signal selection
88  // applied in data reconstruction (see MCH/DigitFiltering/src/DigitFilter.cxx).
89  // a realistic estimate of nSamples would require a complete simulation of the electronic signal
90  double signalParam[3] = {14., 13., 1.5};
91  unsigned int nSamples = std::round(std::pow(double(adc) / signalParam[1], 1. / signalParam[2]) + signalParam[0]);
92  return std::min(nSamples, 0x3FFU); // the number of samples must fit within 10 bits
93 }
